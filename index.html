// 한글과 숫자가 섞인 문자를 숫자로 변환하는 마법의 함수 function parseKoreanAmount(text) { if (!text) return 0; // 1. 쉼표(,)나 '원' 글자 제거 let clean = text.replace(/,/g, '').replace(/원/g, '').trim(); // 2. 순수 숫자만 있는 경우 처리 if (!isNaN(clean) && clean !== "") return Number(clean); // 3. '천', '만', '억' 단위 변환 로직 let total = 0; let temp = ""; // 단위를 숫자로 변환하기 위한 매핑 const units = { '천': 1000, '만': 10000, '백만': 1000000, '천만': 10000000, '억': 100000000 }; // 예: "2천5백만" 같은 복합 입력 처리 (간단 버전) // "2천" -> 2 * 1000, "3만" -> 3 * 10000 if (clean.includes('억')) { let parts = clean.split('억'); total += (Number(parts[0]) || 0) * 100000000; clean = parts[1] || ""; } if (clean.includes('만')) { let parts = clean.split('만'); total += (Number(parts[0]) || 0) * 10000; clean = parts[1] || ""; } if (clean.includes('천')) { let parts = clean.split('천'); total += (Number(parts[0]) || 0) * 1000; } return total || Number(clean.replace(/[^0-9]/g, '')); } // 입력 시 실시간으로 호출되는 함수 (기존 calc 함수 수정) function calc() { let total = 0; document.querySelectorAll('.bet').forEach(input => { const val = parseKoreanAmount(input.value); total += val; // 입력창 옆에 숫자로 변환된 결과를 작게 보여주면 학생들이 더 안심합니다. const previewId = "preview-" + input.dataset.team; let preview = document.getElementById(previewId); if(!preview) { preview = document.createElement('span'); preview.id = previewId; preview.style.fontSize = "0.7em"; preview.style.color = "#888"; input.parentNode.insertBefore(preview, input.nextSibling); } preview.innerText = val > 0 ? ` (${val.toLocaleString()}원)` : ""; }); const btn = document.getElementById('submitBtn'); const status = document.getElementById('status'); if(total === LIMIT) { status.innerText = "✅ 금액이 정확합니다!"; status.style.color = "green"; btn.disabled = false; } else { const diff = LIMIT - total; status.innerText = diff > 0 ? `⚠️ ${diff.toLocaleString()}원 더 투자하세요.` : `❌ ${Math.abs(diff).toLocaleString()}원 초과했습니다!`; status.style.color = diff > 0 ? "orange" : "red"; btn.disabled = true; } }
